Table Usuarios {
  id_usuario int [primary key]
  nombre varchar(50)
  correo_electronico varchar(50)
  contrasena varchar(50)
}

Table Pacientes {
  id_paciente int [primary key]
  id_usuario int [unique]
  genero varchar(10)
  edad int
  tipo_sangre varchar(5)
  ultima_transfusion date
  historial_medico text
}

Table Donantes {
  id_donante int [primary key]
  id_usuario int [unique]
  genero varchar(10)
  edad int
  tipo_sangre varchar(5)
  ultima_donacion date
  historial_medico text
}

Table UsuariosHospital {
  id_usuario_hospital int [primary key]
  id_usuario int [unique]
  id_hospital int
}

Table Administradores {
  id_administrador int [primary key]
  id_usuario int [unique]
  id_hospital int
  id_banco_sangre int
}

Table BancosDeSangre {
  id_banco_sangre int [primary key]
  nombre varchar(50)
  direccion varchar(100)
  telefono varchar(15)
}

Table Hospitales {
  id_hospital int [primary key]
  nombre varchar(50)
  direccion varchar(100)
  telefono varchar(15)
}

Table CampanasDeSangre {
  id_campana_sangre int [primary key]
  nombre varchar(50)
  fecha_inicio date
  fecha_fin date
  id_banco_sangre int
}

Table DonacionesDeSangre {
  id_donacion_sangre int [primary key]
  id_usuario int
  id_campana_sangre int
  fecha_donacion date
}

Table InformacionMedica {
  id_informacion_medica int [primary key]
  id_paciente int
  historial_medico text
}

Table PersonalHospital {
  id_usuario_hospital int
  id_hospital int
  primary key (id_usuario_hospital, id_hospital)
}
